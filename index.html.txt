const clientId = "YOUR_CLIENT_ID";

const content = document.getElementById("content");
const miniPlayer = document.getElementById("miniPlayer");

const btnHome = document.getElementById("btnHome");
const btnSearch = document.getElementById("btnSearch");
const btnLibrary = document.getElementById("btnLibrary");

// ----- Функция подсветки активной вкладки -----
function setActive(button){
    [btnHome, btnSearch, btnLibrary].forEach(b => b.classList.remove("active"));
    button.classList.add("active");
}

// ----- Главная (Trending) -----
function showHome(){
    setActive(btnHome);
    content.innerHTML = "<h2>Trending</h2>";
    fetch(`https://api-v2.soundcloud.com/tracks?client_id=${clientId}&limit=10`)
    .then(res=>res.json())
    .then(data=>{
        data.forEach(track => addTrack(track));
    });
}

// ----- Поиск -----
function showSearch(){
    setActive(btnSearch);
    content.innerHTML = `
        <h2>Search</h2>
        <input id="searchInput" placeholder="Search...">
        <div id="results"></div>
    `;
    document.getElementById("searchInput")
    .addEventListener("keypress", function(e){
        if(e.key === "Enter"){
            searchTracks(this.value);
        }
    });
}

function searchTracks(query){
    fetch(`https://api-v2.soundcloud.com/search/tracks?q=${query}&client_id=${clientId}&limit=10`)
    .then(res=>res.json())
    .then(data=>{
        const results = document.getElementById("results");
        results.innerHTML="";
        data.collection.forEach(track => addTrack(track, results));
    });
}

// ----- Библиотека / Избранное -----
function showLibrary(){
    setActive(btnLibrary);
    content.innerHTML = "<h2>Favorites</h2>";
    const favs = JSON.parse(localStorage.getItem("favorites")) || [];
    favs.forEach(track => addTrack(track));
}

// ----- Добавление трека -----
function addTrack(track, container=content){
    const div = document.createElement("div");
    div.className="track";
    div.innerHTML = `
        ${track.title} - ${track.user.username} 
        <span style="float:right;">❤️</span>
    `;
    div.onclick = ()=>playTrack(track);
    div.querySelector("span").onclick = (e)=>{
        e.stopPropagation();
        saveFavorite(track);
    };
    container.appendChild(div);
}

// ----- Мини-плеер -----
function playTrack(track){
    miniPlayer.innerHTML = `
        <iframe width="100%" height="120" scrolling="no" frameborder="no"
        src="https://w.soundcloud.com/player/?url=${track.permalink_url}&auto_play=true&color=%231db954">
        </iframe>
    `;
}

// ----- Сохранение избранного -----
function saveFavorite(track){
    let favs = JSON.parse(localStorage.getItem("favorites")) || [];
    if(!favs.some(t => t.id === track.id)){
        favs.push(track);
        localStorage.setItem("favorites", JSON.stringify(favs));
        alert("Added to Favorites!");
    }
}

// ----- Запуск -----
showHome();
